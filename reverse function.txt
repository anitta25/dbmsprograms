declare
  n integer :=321;
  d integer;
  s integer:=0;

 function reverse(n in out integer) return integer as
 begin
   
   while n>0 loop
   
    d:=mod(n,10);
    s:=s*10+d;

    n:=n/10;
   end loop;
   return s;
   end ;
   begin
   dbms_output.put_line(reverse(n));
   end;
